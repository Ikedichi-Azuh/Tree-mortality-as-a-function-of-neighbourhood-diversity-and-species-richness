3.7 Function to replace values after consecutive NAs with NA

# Replace values that followed two or more consecutive 
update_na_values <- function(group) {
  n <- nrow(group)
  heights <- group$Height
  diameters <- group$Diameter
  
  # Find positions of NAs in Height
  na_positions <- which(is.na(heights))
  
  if (length(na_positions) == 0 || length(na_positions) == n) {
    return(group)  # No NA or all NA, return unchanged
  }
  
  # Iterate through the positions of NAs
  for (i in seq_along(na_positions)) {
    pos <- na_positions[i]
    
    if (pos > 1 && pos < n) {  # Ensure it's between values
      prev_value <- heights[pos - 1]
      next_pos <- pos
      
      # Move forward to find the end of the NA sequence
      while (next_pos < n && is.na(heights[next_pos])) {
        next_pos <- next_pos + 1
      }
      
# If next value is non-NA and the previous value is also non-NA, then we have a bounded NA sequence
      if (next_pos <= n && !is.na(heights[next_pos]) && !is.na(prev_value)) {
        # Replace all values after the NA sequence with NA in both Height and Diameter
        heights[next_pos:n] <- NA
        diameters[next_pos:n] <- NA
        break  # Stop after first bounded sequence
      }
    }
  }
  
  group$Height <- heights
  group$Diameter <- diameters
  return(group)
}

3.8 Create “Status” column using the following conditions:

If for a tree:

  1. Only first 2 years have values ~ “D”
  2. Only first year has value ~ “C”
  3. Only second, third, or fourth year has a value ~ “Delete”
  4. All entries are NAs ~ “Delete”
  5. Maximum year is NA ~ “D”
  6. Maximum year has values ~ “A”
library(dplyr)

raw_df6 <- raw_df5 %>%
  arrange(Tree_ID, Year) %>%
  group_by(Tree_ID) %>%
  mutate(
    # Count non-NA values for each ID
  non_na_count = sum(!is.na(Height)),
    
    # Find years where measurements exist (store as a list)
  years_with_values = list(Year[!is.na(Height)]),
    
    # Identify first and last years with measurements safely
  first_year = ifelse(length(years_with_values[[1]]) > 0, min(years_with_values[[1]], na.rm = TRUE), NA),
  last_year = ifelse(length(years_with_values[[1]]) > 0, max(years_with_values[[1]], na.rm = TRUE), NA),
    
    # Identify if only the first two years (2014, 2015) have data
  first_two_only = length(years_with_values[[1]]) == 2 && all(years_with_values[[1]] %in% c(2014, 2015)),
    
    
    # Assign Status based on conditions **without creating extra columns**
  Status = case_when(
    non_na_count == 0 ~ "Delete",  # All values are NA
    non_na_count == 1 & first_year %in% c(2015, 2016, 2017, 2018) ~ "Delete",  # Only 2/3/4th year
    non_na_count == 1 & first_year == 2014 ~ "C",                         # Only 1st year has value
    first_two_only ~ "D",                           # Only first two years (2014, 2015) have values
    last_year == 2023 ~ "A",                               # If last year has values, Status is "A"
    last_year < 2023 ~ "D",          # If max year is NA or missing data before 2023, Status is "D"
    TRUE ~ NA_character_             # Default case
    )  %>%
      as.factor()                    # Convert to factor within mutate
  ) %>%
  ungroup() %>%
  dplyr::select(-non_na_count, -years_with_values, -first_year, -last_year,
         -first_two_only)            # Remove helper columns

5 Computing Heygi’s competition index (1974)

Here we use diameter as measure of size and competition radius of 1,2 and 3.

library(sf)

library(units)

# Convert the dataset into an sf object using X and Y as coordinates
tree_sf <- st_as_sf(raw_df10, coords = c("X", "Y"), crs = 32633)

# Unique years in the dataset
years <- unique(tree_sf$Year)

# Competition radii to loop over
competition_radii <- c(1, 2, 3)

# Initialize a list to store CI results for each year
ci_results <- list()

# Loop through each year and calculate CI for each radius and each tree
for (year in years) {
  
  # Filter data for the current year
  year_data <- tree_sf %>% filter(Year == year)
  
  # Calculate pairwise distances for the current year
  dist_matrix <- st_distance(year_data)
  dist_matrix <- set_units(dist_matrix, NULL)  # Remove units for easier computation
  
  # Loop through each competition radius
  for (radius in competition_radii) {
    
    # Function to calculate CI for each tree (Heygi's Competition Index)
    calculate_ci <- function(i, diameter, dist_matrix, radius) {
      distances <- dist_matrix[i, ]
      distances <- as.numeric(distances)
      
      # Identify neighbors within the competition radius (excluding the focal tree itself)
      within_radius <- which(distances <= radius & distances > 0)
      
      # Filter out neighbors with missing heights
      valid_neighbors <- within_radius[!is.na(diameter[within_radius])]
      
      if (length(valid_neighbors) == 0) {
        return(0)  # No valid neighbors within radius
      }
      
      # Get heights and distances for valid neighbors
      diameters_within_radius <- diameter[valid_neighbors]
      distances_within_radius <- distances[valid_neighbors]
      
      # Calculate the CI value for the target tree i (HCI)
      ci_value <- sum(diameters_within_radius / (diameter[i] * distances_within_radius))
      return(ci_value)
    }
    
    # Apply CI calculation to each tree for the given radius
    year_data[[paste0("CI_", radius)]] <- sapply(1:nrow(year_data), 
                                                 function(i) calculate_ci(i, year_data$Diameter, 
                                                                          dist_matrix, radius))
  }
  
  # Store results for this year
  ci_results[[as.character(year)]] <- year_data
}

# Combine the results into a single data frame
ci_combined <- do.call(rbind, ci_results)

############################################
# Calculate the median CI for each of the CI columns
############################################
ci_medians <- sapply(competition_radii, function(radius) {
  median(ci_combined[[paste0("CI_", radius)]], na.rm = TRUE)
})

############################################
# Classify each CI column as "Dominant" or "Subordinant"
############################################
for (radius in competition_radii) {
  ci_column <- paste0("CI_", radius)
  cat_column <- paste0(ci_column, "_cat")
  
  ci_combined[[cat_column]] <- ifelse(ci_combined[[ci_column]] <= ci_medians[[radius]],
                                      "Dominant", "Subordinant")
}

df_ci_combined <- as.data.frame(ci_combined)

remove(tree_sf,years,competition_radii,ci_combined,ci_medians,ci_results)
